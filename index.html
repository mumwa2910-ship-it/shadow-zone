<script>
const c = document.getElementById('game');
const ctx = c.getContext('2d');
c.width = innerWidth; c.height = innerHeight;

// ===== DATA SYSTEM =====
let save = JSON.parse(localStorage.getItem('sz_save')) || {
  level:1,xp:0,rp:0,rank:'Bronze I',coins:500,diamonds:0,
  ownedChars:['Rook'],ownedSkins:['AK_Default'],equipChar:'Rook',equipSkin:'AK_Default'
};
function saveGame(){localStorage.setItem('sz_save',JSON.stringify(save))}
function showToast(t){const el=document.getElementById('toast');el.textContent=t;el.style.display='block';setTimeout(()=>el.style.display='none',1500)}

// ===== XP/RANK SYSTEM =====
const rankOrder=['Bronze I','Bronze II','Bronze III','Silver I','Silver II','Gold I','Gold II','Platinum','Diamond','Heroic','Grandmaster'];
function addXP(amt){
  save.xp+=amt;
  let need=100+save.level*50;
  if(save.xp>=need){save.xp-=need;save.level++;save.coins+=100;showToast(`LEVEL UP! Lv.${save.level}`)}
  updateUI();
}
function addRP(amt){
  save.rp+=amt;
  if(save.rp>=100){save.rp=0;let i=rankOrder.indexOf(save.rank);if(i<rankOrder.length-1){save.rank=rankOrder[i+1];save.coins+=300;showToast(`RANK UP! ${save.rank}`)}}
  if(save.rp<0)save.rp=0;
  updateUI();
}
function updateUI(){
  document.getElementById('coins').textContent=save.coins;
  document.getElementById('diamonds').textContent=save.diamonds;
  document.getElementById('ranktxt').textContent=`${save.rank} | Lv.${save.level}`;
  document.getElementById('xpfill').style.width=(save.xp/(100+save.level*50))*100+'%';
}

// ===== SHOP DATA =====
const chars=[
  {id:'Rook',name:'Rook',cost:0,ad:0,desc:'Default'},
  {id:'Specter',name:'Specter',cost:1500,ad:1,desc:'15% Faster Reload'},
  {id:'Blaze',name:'Blaze',cost:2000,ad:2,desc:'+20 Armor Start'},
  {id:'Phantom',name:'Phantom',cost:9999,ad:3,desc:'Quieter Steps'}
];
const skins=[
  {id:'AK_Default',name:'AK Default',gun:'AK',cost:0,ad:0},
  {id:'AK_Desert',name:'Desert Camo',gun:'AK',cost:1000,ad:0},
  {id:'M4_Neon',name:'Neon Red',gun:'M4',cost:1200,ad:1},
  {id:'AWM_Gold',name:'Gold Dragon',gun:'AWM',cost:9999,ad:3}
];

// ===== GAME STATE =====
let state='lobby',player,bots=[],bullets=[],zone={x:0,y:0,r:800,maxR:800,shrinkTimer:60,damage:5},keys={},mouse={x:0,y:0,down:false};
let mapSize=2000,hardMode=false;

// ===== INPUT =====
addEventListener('keydown',e=>keys[e.key.toLowerCase()]=true);
addEventListener('keyup',e=>keys[e.key.toLowerCase()]=false);
document.getElementById('fire').ontouchstart=()=>mouse.down=true;
document.getElementById('fire').ontouchend=()=>mouse.down=false;
document.getElementById('crouch').onclick=()=>{if(player)player.state=player.state==='crouch'?'stand':'crouch'}
document.getElementById('prone').onclick=()=>{if(player)player.state=player.state==='prone'?'stand':'prone'}
document.getElementById('slide').onclick=()=>{if(player&&!player.slideCD)player.slide()}

// Joystick
let joyStart={x:0,y:0};
document.getElementById('joystick').ontouchstart=e=>{joyStart={x:e.touches[0].clientX,y:e.touches[0].clientY}}
document.getElementById('joystick').ontouchmove=e=>{
  let dx=e.touches[0].clientX-joyStart.x,dy=e.touches[0].clientY-joyStart.y;
  let mag=Math.min(Math.hypot(dx,dy),60);
  let ang=Math.atan2(dy,dx);
  document.getElementById('stick').style.left=35+Math.cos(ang)*mag+'px';
  document.getElementById('stick').style.top=35+Math.sin(ang)*mag+'px';
  if(player){player.vx=Math.cos(ang)*mag/60;player.vy=Math.sin(ang)*mag/60}
document.getElementById('joystick').ontouchend=()=>{
  document.getElementById('stick').style.left='35px';document.getElementById('stick').style.top='35px';
  if(player){player.vx=0;player.vy=0}
}

// ===== PLAYER CLASS =====
class Player{
  constructor(x,y,bot=false){
    this.x=x;this.y=y;this.hp=100+(save.equipChar==='Blaze'?20:0);
    this.vx=0;this.vy=0;this.state='stand';this.slideCD=0;this.bot=bot;
    this.speed=2.5;this.size=18;this.kills=0;this.reload=0;
  }
  update(){
    let speed=this.speed*(this.state==='crouch'?0.7:this.state==='prone'?0.4:1)*(keys['shift']?1.5:1);
    if(!this.bot){
      if(keys['w'])this.vy=-speed;
      if(keys['s'])this.vy=speed;
      if(keys['a'])this.vx=-speed;
      if(keys['d'])this.vx=speed;
    }
    this.x+=this.vx;this.y+=this.vy;
    this.x=Math.max(20,Math.min(mapSize-20,this.x));
    this.y=Math.max(20,Math.min(mapSize-20,this.y));
    this.vx*=0.8;this.vy*=0.8;
    if(this.slideCD>0)this.slideCD--;
    if(this.reload>0)this.reload--;

    // Zone damage
    let d=Math.hypot(this.x-zone.x,this.y-zone.y);
    if(d>zone.r)this.hp-=zone.damage/60;

    // Bot AI
    if(this.bot){
      let target=player;
      let ang=Math.atan2(target.y-this.y,target.x-this.x);
      this.vx=Math.cos(ang)*speed*0.9;
      this.vy=Math.sin(ang)*speed*0.9;
      if(Math.random()<0.02&&hardMode)this.state=Math.random()>.5?'crouch':'prone';
      if(Math.hypot(target.x-this.x,target.y-this.y)<400&&this.reload<=0)this.shoot(ang);
    }
  }
  shoot(ang){
    if(this.reload>0)return;
    bullets.push({x:this.x,y:this.y,vx:Math.cos(ang)*10,vy:Math.sin(ang)*10,owner:this});
    this.reload=save.equipChar==='Specter'?8:10;
  }
  slide(){this.slideCD=60;this.vx=-this.vx*3;this.vy=-this.vy*3}
  draw(){
    ctx.fillStyle=this.bot?'#888':'#0af';
    if(this.state==='prone')ctx.fillRect(this.x-12,this.y-6,24,12);
    else if(this.state==='crouch')ctx.fillRect(this.x-10,this.y-12,20,20);
    else ctx.fillRect(this.x-this.size,this.y-this.size,this.size*2,this.size*2);
    ctx.fillStyle='#f00';ctx.fillRect(this.x-15,this.y-22,this.hp/100*30,4);
  }
}

// ===== GAME LOOP =====
function loop(){
  if(state==='game'){
    ctx.clearRect(0,0,c.width,c.height);
    ctx.save();
    ctx.scale(c.width/mapSize,c.height/mapSize);
    ctx.translate(-player.x+c.width/2/c.width*mapSize,-player.y+c.height/2/c.height*mapSize);

    // Map
    ctx.fillStyle='#2a3f2a';ctx.fillRect(0,0,mapSize,mapSize);

    // Zone visual
    ctx.strokeStyle='red';ctx.lineWidth=5;ctx.beginPath();ctx.arc(zone.x,zone.y,zone.r,0,6.28);ctx.stroke();

    player.update();player.draw();
    bots.forEach(b=>{b.update();b.draw()});
    bullets.forEach((bu,i)=>{
      bu.x+=bu.vx;bu.y+=bu.vy;
      ctx.fillStyle='#ff0';ctx.fillRect(bu.x-3,bu.y-3,6,6);
      [...bots,player].forEach(t=>{
        if(t!==bu.owner&&Math.hypot(t.x-bu.x,t.y-bu.y)<15){
          t.hp-=20;bullets.splice(i,1);
          if(t.hp<=0&&t.bot){player.kills++;addXP(10);addRP(3);save.coins+=10}
        }
      })
    });

    ctx.restore();
    document.getElementById('hp').textContent=Math.max(0,Math.floor(player.hp));
    document.getElementById('kills').textContent=player.kills;
    document.getElementById('zonetimer').textContent=Math.floor(zone.shrinkTimer);

    zone.shrinkTimer-=1/60;
    if(zone.shrinkTimer<=0){zone.r*=0.85;zone.shrinkTimer=hardMode?40:60;zone.damage+=2}

    if(player.hp<=0||bots.every(b=>b.hp<=0))endMatch();
  }
  requestAnimationFrame(loop);
}

// ===== MODES =====
function startBR(){
  hardMode=rankOrder.indexOf(save.rank)>=rankOrder.indexOf('Gold I');
  player=new Player(mapSize/2,mapSize/2);
  bots=Array.from({length:30},()=>new Player(Math.random()*mapSize,Math.random()*mapSize,true));
  zone={x:mapSize/2,y:mapSize/2,r:800,maxR:800,shrinkTimer:60,damage:5};
  state='game';showScreen('gameui');document.getElementById('minimap').classList.remove('hidden');document.getElementById('zone').classList.remove('hidden');
}
function startClash(){showToast('Clash Squad: V2 coming next update')}
function start1v1(){showToast('1v1 Room: V2 coming next update')}
function endMatch(){
  let won=bots.every(b=>b.hp<=0);
  addXP(won?25:5);addRP(won?30:-5);save.coins+=won?50:15;
  saveGame();showScreen('lobby');state='lobby';
  showToast(won?'VICTORY!':'DEFEATED');
}

// ===== SHOP UI =====
function renderShop(){
  let html='';
  [...chars,...skins].forEach(it=>{
    let owned=save.ownedChars.includes(it.id)||save.ownedSkins.includes(it.id);
    html+=`<div class="item"><div><b>${it.name}</b><br><small>${it.desc||it.gun}</small></div>
    <div>${owned?'<button disabled>OWNED</button>':
    `<button class="small" onclick="buy('${it.id}',${it.cost})">${it.cost}💰</button>
     <button class="adbtn" onclick="watchAd('${it.id}',${it.ad})">AD ${it.ad}</button>`}</div></div>`
  });
  document.getElementById('shopitems').innerHTML=html;
}
function buy(id,cost){if(save.coins>=cost){save.coins-=cost;save.ownedChars.push(id);save.ownedSkins.push(id);saveGame();renderShop();updateUI()}}
function watchAd(id,ads){if(ads>0){let t=ads*5;let i=setInterval(()=>{t--;if(t<=0){clearInterval(i);save.diamonds++;save.ownedChars.push(id);save.ownedSkins.push(id);saveGame();renderShop();updateUI();showToast('Unlocked!')}},1000);showToast(`Watching Ad... ${ads*5}s`)}}

// ===== EQUIP UI =====
function renderEquip(){
  let html='<h3>Character</h3>';
  chars.filter(c=>save.ownedChars.includes(c.id)).forEach(c=>html+=`<button class="small" onclick="save.equipChar='${c.id}';saveGame();renderEquip()">${c.name} ${save.equipChar===c.id?'[E]'}</button>`);
  html+='<h3>Skin</h3>';
  skins.filter(s=>save.ownedSkins.includes(s.id)).forEach(s=>html+=`<button class="small" onclick="save.equipSkin='${s.id}';saveGame();renderEquip()">${s.name} ${save.equipSkin===s.id?'[E]'}</button>`);
  document.getElementById('equiplist').innerHTML=html;
}

// ===== UI HELPERS =====
function showScreen(id){
  document.querySelectorAll('.screen').forEach(s=>s.classList.add('hidden'));
  document.getElementById(id).classList.remove('hidden');
  if(id==='shop')renderShop();
  if(id==='equip')renderEquip();
  if(id==='gameui'){}else document.getElementById('gameui').classList.add('hidden');
}

// INIT
updateUI();loop();
</script>
